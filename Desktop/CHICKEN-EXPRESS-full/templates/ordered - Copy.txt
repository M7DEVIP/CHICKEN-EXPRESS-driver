    function searchOrders() {
        var input = searchInputOrder.value.toUpperCase();

        cards.forEach(function(card) {

                card.style.display = "none";
            }
        });
    }

    function toggleAddButtonNew(button) {
        button.classList.add("hidden");
        var quantityDiv = button.nextElementSibling;
        var quantitySpan = quantityDiv.querySelector("span");
        quantitySpan.innerHTML = "1";
        quantityDiv.classList.remove("hidden");
        var quantityButtons = quantityDiv.getElementsByTagName("button");
        for (var i = 0; i < quantityButtons.length; i++) {
            quantityButtons[i].addEventListener("click", handleQuantityButtonClickNew);
        }
        updateOrderSummaryNew();
    }

    function handleQuantityButtonClickNew(event) {
        var button = event.target;
        var quantitySpan = button.parentNode.querySelector("span");
        var quantity = parseInt(quantitySpan.innerHTML);

        if (button.classList.contains("quantity-button")) {
            if (button.innerHTML === "+") {
                quantity++;
            } else if (button.innerHTML === "-") {
                if (quantity > 0) {
                    quantity--;
                }
            }
            quantitySpan.innerHTML = quantity;
        }

        if (quantity === 0 && !button.classList.contains("add-button")) {
            var quantityDiv = button.parentNode;
            quantityDiv.classList.add("hidden");
            var addButton = quantityDiv.parentNode.querySelector(".add-button");
            addButton.classList.remove("hidden");
        }

        updateOrderSummaryNew();
    }

    var baseTotalNew = 0; // متغير لحفظ السعر الأساسي الإجمالي

    function updateOrderSummaryNew() {
        var orderList = document.getElementById("order-list-new");
        var totalPriceSpan = document.querySelector("#total-price-new span");
        var totalPriceInput = document.getElementById("total-price-input-new");

        orderList.innerHTML = "";
        var menuItems = document.querySelectorAll(".menu-item");
        baseTotalNew = 0; // إعادة تعيين السعر الأساسي الإجمالي
        var orderDetailsArray = [];

        menuItems.forEach(function(menuItem) {
            var quantitySpan = menuItem.querySelector(".quantity span");
            var quantity = parseInt(quantitySpan.innerHTML);

            if (quantity > 0) {
                var itemName = menuItem.querySelector("h2").innerHTML;
                var price = parseFloat(menuItem.querySelector("p:nth-child(3)").innerHTML.split(': ')[1]);
                if (isNaN(price)) price = 0;
                var itemTotalPrice = quantity * price;

                var listItem = document.createElement("li");
                listItem.innerHTML = quantity + " x " + itemName + " (" + itemTotalPrice.toFixed(2) + " دينار)";
                orderList.appendChild(listItem);

                baseTotalNew += itemTotalPrice; // تحديث السعر الأساسي الإجمالي
                orderDetailsArray.push(quantity + " x " + itemName + " (" + itemTotalPrice.toFixed(2) + " دينار)");
            }
        });

        totalPriceSpan.innerHTML = baseTotalNew.toFixed(2);
        totalPriceInput.value = baseTotalNew.toFixed(2); // تحديث المدخل للسعر الأساسي الإجمالي
        document.getElementById("order-details-new").value = orderDetailsArray.join("\n");
    }

    function updateFinalPriceNew() {
        var finalPrice = baseTotalNew; // استخدام السعر الأساسي الإجمالي للحسابات

        // حساب تكلفة المنطقة
        var zone = document.getElementById("zone-new").value;
        if (zone === "داخل الزون") {
            finalPrice += 2000;
        } else if (zone === "خارج الزون") {
            finalPrice += 3000;
        }

        // حساب الخصم
        var discount = parseFloat(document.getElementById("discount-new").value);
        if (isNaN(discount)) discount = 0;
        var discountAmount = (finalPrice * discount) / 100;
        finalPrice -= discountAmount;

        if (isNaN(finalPrice)) finalPrice = 0;

        document.getElementById("total-price-input-new").value = finalPrice.toFixed(2);
    }

    document.getElementById("zone-new").addEventListener("change", updateFinalPriceNew);

                    success: function(response) {
                        $('#name-new').val(response.Cname);
                        $('#address-new').val(response.Caddress);
                        $('#notes-new').val(response.Cnote);
                    },
                    error: function() {
                        $('#name-new').val('');
                        $('#address-new').val('');
                        $('#notes-new').val('');
                    }
                });
            } else {
                $('#name-new').val('');
                $('#address-new').val('');
                $('#notes-new').val('');
            }
        });

        $('#order-form-new').on('submit', function(event) {
            event.preventDefault();
            var orderDetails = $('#order-details-new').val();
            var phone = $('#phone-new').val();
            var name = $('#name-new').val();
            var address = $('#address-new').val();
            var notes = $('#notes-new').val();
            var zone = $('#zone-new').val();
            var discount = $('#discount-new').val();
            var tocost = parseFloat($('#total-price-input-new').val());
            if (isNaN(tocost)) tocost = 0;

            var submitButton = $(this).find('button[type="submit"]');
            submitButton.css('background-color', '#f39c12');

            $.ajax({
                url: '/make-order',
                type: 'POST',
                data: {
                    order: orderDetails,
                    phone: phone,
                    name: name,
                    address: address,
                    note: notes,
                    zone: zone,
                    discount: discount,
                    tocost: tocost
                },
                success: function(response) {
                    alert(response);
                    $('#order-form-new')[0].reset();
                    submitButton.css('background-color', '');
                    document.body.classList.remove("no-scroll");
                    document.getElementById("form-overlay-new").classList.add("hidden");
                    document.getElementById("order-form-new").classList.add("hidden");
                    document.querySelectorAll("button, input, textarea").forEach(function(element) {
                        element.disabled = false;
                    });
                    window.location.reload();
                },
                error: function() {
                    alert('حدث خطأ أثناء إرسال الطلب');
                    submitButton.css('background-color', '');
                }
            });
        });

        updateOrderSummaryNew(); // تحديث ملخص الطلب عند تحميل الصفحة
    });
